package ar.com.unpaz.serviceImpl;

import java.util.List;


import ar.com.unpaz.model.*;
import ar.com.unpaz.service.GestionEmpleadoService;

public class GestionEmpleadoServiceImpl implements GestionEmpleadoService{

	@Override
	public void agregarEmpleado(List<Admin> empleados, Admin empleado) {
		empleados.add(empleado);
		System.out.println("Empleado agregado");
		
		
		
	}

	@Override
	public void buscarEmpleado(List<Admin> empleados, String legajo) {
		boolean encontrado = false;
		for(Admin empleado: empleados) {
			if(empleado.getLegajo().equals(legajo)) {
				System.out.println(empleado);
				encontrado = true;
				break;
				
			}
		}
		if(!encontrado) {
			System.out.println("No se encontro ese empleado");
		}
		
	}

	@Override
	public void mostrarSueldo(List<Admin> empleados, double sueldo) {
		for(Admin empleado: empleados) {
			double total = empleado.calcularSueldo();
			System.out.println("Empleado: " + empleado.getNombre() + "Sueldo total: " + total);
		}
		
	}

    @Override
    public void darBaja(List<Admin> empleados, String legajo) {

        boolean encontrado = false;

        for (Admin empleado : empleados) {
            if (empleado.getLegajo().equalsIgnoreCase(legajo)) {
            	empleados.remove(empleado);
                System.out.println("Empleado dado de baja");
                encontrado = true;
                break;
            }
        }

        if (!encontrado) {
            System.out.println("Empleado no encontrado");
        }
    }


	@Override
	public void cantidadInactivos(List<Admin> empleados, boolean activo) {
		int contador = 0;
		for(Admin empleado: empleados) {
			if(empleado.isActivo() == !activo) {
				contador++;
			}
		}
		if(contador > 0) {
			System.out.println("Cantidad de inactivos: " + contador);
			
		} else {
			System.out.println("No hay incativos");
			
		}
		
		
	}

	@Override
	public void mayorSueldo(List<Admin> empleados, double sueldo) {
		Empleado mayorSueldo = empleados.get(0);
		for(Admin empleado: empleados) {
			if(empleado.calcularSueldo() > mayorSueldo.getSueldo()) {
				mayorSueldo = empleado;
			}
		}
		System.out.println("Mayor Sueldo: " + mayorSueldo);
		
		
	}

	@Override
	public void listarEmpleados(List<Admin> empleados) {
		if(empleados.isEmpty()) {
			System.out.println("No hay empleados");
			
		}else {
			for(Admin empleado: empleados) {
				System.out.println(empleado);
				
			}
			
		}
		
	}

}
